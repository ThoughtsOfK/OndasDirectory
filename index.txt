/* ── Toggle subsidiary accordion ── */
function toggleSub(hdr) {
  var wrap = hdr.closest('.sub-collapsible');
  if (!wrap) return;
  var isOpen = wrap.classList.contains('expanded');
  wrap.classList.toggle('expanded', !isOpen);
}

  wrap.querySelectorAll('.stab').forEach(b => b.classList.remove('active'));
  wrap.querySelectorAll('.sub-pane').forEach(p => p.classList.remove('active'));
  btn.classList.add('active');

  if (pane) pane.classList.add('active');
